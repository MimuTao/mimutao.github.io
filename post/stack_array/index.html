Stack: Use c and array

stack.h

    #ifndef STACK_H
    #define STACK_H
    #include<stdio.h>
    #include<stdbool.h>
    #include<malloc.h>
    #define Max_size 100

    typedef int Element_type;
    typedef struct Node *Stack;

    struct Node{
        Element_type data[Max_size];
        int Top;
    };

    // Init 
    Stack Init(){
        Stack stack = (Stack)malloc(sizeof(struct Node));
        stack->Top = -1;    // use top=-1 to express the stack is empty. 
        return stack;
    }

    // is_empty
    bool is_empty(Stack stack){
        return stack->Top==-1?true:false; 
    }

    // is_full
    bool is_full(Stack stack){
        return stack->Top==Max_size-1?true:false;
    }

    // Push
    void Push(Element_type x, Stack stack){
        if(is_full(stack)){
            printf("The stack is full, have no room to push.\n");
            return;
        }
        ++stack->Top;
        stack->data[stack->Top] = x;
    } 

    // Pop 
    void Pop(Stack stack){
        if(is_empty(stack)){
            printf("The stack is empty, have no element to pop.\n");
            return;
        }
        stack->Top--;
    }

    // The size of the stack.
    int Size(Stack stack){
        return stack->Top+1;
    }

    // Traverse
    void Traverse(Stack stack){
        int i;
        for(i=0;i<=stack->Top;++i){
            printf("The %d element in stack is %d.\n",i+1,stack->data[i]);
        }
    }

    #endif
