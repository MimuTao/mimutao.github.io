Stack: Use c and linklist

stack.h

    #ifndef STACK_H
    #define STACK_H
    #include<stdio.h>
    #include<malloc.h>
    #include<stdbool.h>

    typedef int Element_type;
    typedef struct Node *Stack;

    struct Node{
        Element_type data;
        Stack next;
    };

    // Initilcation.
    Stack Init(){
        Stack stack = (Stack)malloc(sizeof(struct Node));
        stack->next = NULL;
        return stack;
    }

    //// Push
    //Stack Push(Element_type x, Stack stack){

    //  stack_top->data = x;
    //  stack_top->next = stack;
    //  return stack_top;
    //}

    // Push a better way.
    void Push(Element_type x, Stack stack){
        Stack temp = (Stack)malloc(sizeof(struct Node));
        temp->data = x;
        temp->next = stack->next;
        stack->next = temp; 
    }

    //// Pop
    //Stack Pop(Stack stack){
    //  if(stack->next == NULL){
    //      printf("The stack is empty, have no element to pop.\n");
    //      return NULL;
    //  }
    //  Stack temp = stack;
    //  stack = stack->next; 
    //  free(temp);    // release the memery of top element of stack.
    //  return stack;
    //}

    // Pop another way.
    void Pop(Stack stack){
        if(stack->next == NULL){
            printf("The stack is empty, have no element to pop.\n");
            return;
        } 
        Stack temp = stack->next;
        stack->next = temp->next;
        free(temp); 
    }

    // The size of stack
    int Size(Stack stack){
        int i = 0;
        while(stack->next != NULL){
            i++;
            stack = stack->next;
        }
        return i;
    }

    // Judge if the stack is empty.
    bool is_empty(Stack stack){
        return stack->next==NULL?true:false;
    }

    // Traverse
    void Traverse(Stack stack){
        int i=1;
        while(stack->next != NULL){
            printf("The %dth element in the stack is %d.\n",i,stack->data);
            stack = stack->next;
        }
    }

#endif

#### main.c #include “stack.h”

    int main(void){
        printf("INITILATION.\n");
        Stack m_stack = Init();
        printf("The size of stack is %d.\n",Size(m_stack));
        printf("Push some element into stack.\n");
        int i = 0;
        for(i;i<4;++i){
            Push(i,m_stack);
        }
        Traverse(m_stack);

        printf("Pop some elements out of stack.\n");
        for(i=0;i!=2;++i){
            Pop(m_stack);
        }
        printf("The size of stack now is %d.\n",Size(m_stack));
        Traverse(m_stack); 

    }
